\section*{Knapsack}

\fbox{\begin{minipage}{35em}
\textbf{Given:}
\begin{itemize}
	\item weights $w[i] \text{, } i=\overline{1,n}$ 
	\item values $c[i] \text{, } i=\overline{1,n}$ 
	\item $W$ - maximum wight, capacity of teh knapsack 
\end{itemize}

\textbf{Find subset of items I such that:}
\begin{itemize}
	\item The total weight of items in $I$ does not exceed $W$, $\sum_{i \in I} w[i] \leq W$
	\item The total vlue of items in $I$ is maximized $W$, $max_{I \subset \{1, \dots, n\}} \sum_{i \in I} c[i] $
\end{itemize}
\end{minipage}}

\section*{Base knapsack solution}

\begin{itemize}
	\item[-] $dp[i][с]$ - stores the minimal achievable weight for the first $i$ elements with values c.
	\item[-] $0 \leq i \leq n$, $0 \leq c \leq C$ 
	\item[-] dp recalc: \\
	\[
	dp[i][c] =
	\begin{cases} 
	dp[i-1][c], & \text{если } c[i] > c, \\
	\min\left(dp[i-1][c], dp[i-1][c - c[i]] + w[i]\right), & \text{иначе}.
	\end{cases}
	\]

\end{itemize}

\textbf{final asymptotics: $\underline{O}(n * C_{max})$}

\section*{Approximation polynomial knapsack}

\fbox{\begin{minipage}{35em}
\textbf{Given:}
\begin{itemize}
	\item weights $w[i] \text{, } i=\overline{1,n}$ 
	\item values $c[i] \text{, } i=\overline{1,n}$ 
	\item $W$ - maximum weights, capacity of the knapsack 
	\item $\varepsilon$ - acceptable error.
\end{itemize}

\textbf{Find subset of items I such that:}
\begin{itemize}
	\item The total weight of items in $I$ does not exceed $W$, $\sum_{i \in I} w[i] \leq W$
	\item Maximum the total vlue of items in $I$ differs from the real answer by no more than $1+\varepsilon$
	\item The algorithm works in polynomial asymptotics.
\end{itemize}
\end{minipage}}

\section*{Approximated polynomial solution}

\begin{enumerate}
	\item Find $K = \frac{\varepsilon * C_{max}}{n}$, $C_{max} = max_{i=\overline{1,n}} c[i]$
	\item $c^{'}[i] = \lfloor{\frac{c[i]}{K}} \rfloor$
	\item Solve based knapsack problem with $c^{'}[i]$
\end{enumerate}

\textbf{final asymptotics: $\underline{O}(n^2 \lfloor \frac{C_{max}}{K} \rfloor) = \underline{O}(n^2 \lfloor \frac{n}{\varepsilon} \rfloor)$}

\textbf{Correct:} 
\url{https://math.mit.edu/~goemans/18434S06/knapsack-katherine.pdf}.
